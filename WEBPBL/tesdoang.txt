class Pekerja:
    def __init__(self, employee_id, salary, office_location):
        self.employee_id = employee_id
        self.salary = salary
        self.office_location = office_location

    def display_info(self):
        print(f"ID: {self.employee_id}")
        print(f"Gaji: {self.salary}")
        print(f"Kantor: {self.office_location}")

class Karyawan(Pekerja):
    def __init__(self, employee_id, salary, office_location):
        super().__init__(employee_id, salary, office_location)

class Manajer(Pekerja):
    def __init__(self, employee_id, salary, office_location, department):
        super().__init__(employee_id, salary, office_location)
        self.department = department

    def display_info(self):
        super().display_info()
        print(f"Departemen: {self.department}")

class Programmer(Pekerja):
    def __init__(self, employee_id, salary, office_location, programming_language):
        super().__init__(employee_id, salary, office_location)
        self.programming_language = programming_language

    def display_info(self):
        super().display_info()
        print(f"Bahasa Pemrograman: {self.programming_language}")

# Contoh penggunaan
karyawan1 = Karyawan("K001", 50000, "Kantor A")
manajer1 = Manajer("M001", 75000, "Kantor B", "Keuangan")
programmer1 = Programmer("P001", 60000, "Kantor C", "Python")

print("Info Karyawan:")
karyawan1.display_info()

print("\nInfo Manajer:")
manajer1.display_info()

print("\nInfo Programmer:")
programmer1.display_info()
